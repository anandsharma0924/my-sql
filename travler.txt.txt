create database StarTravels;
use StarTravels;
create table AdminData(AdminId int , AdminPass varchar(100));
create table UserData(UserId int auto_increment , UserName varchar(300),UserAge int , UserEmail varchar(300),UserPass varchar(350),primary key(UserId))auto_increment=1000;
create table BusData(BusNumber varchar(300),StartLocation varchar(350), EndLocation varchar(350),DepartureTime time , ArrivalTime time,ReturnDepartureTime time , ReturnArrivalTime time , TotalSeat int , SeatPrice int);
create table BookingData(BookingId int auto_increment , userId int , BusNumber varchar(300),FromLocation varchar(300) , ToLocation varchar(300),BookingDate date,TotalSeat int , TotalAmount int , primary key(BookingId))auto_increment=1;
create table FeedBackData(FeedBackId int auto_increment , userId int , BusNumber varchar(300),Rating int,primary key(FeedBackId))auto_increment=1;
